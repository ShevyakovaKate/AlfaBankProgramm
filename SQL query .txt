1) SELECT DISTINCT NAME, POSITION
   FROM EMPLOYEE
   WHERE DEPARTMENT_ID = NAME_OF_DEPARTMENT

2) SELECT b.NAME
   FROM (SELECT CHIEF_ID, SALARY FROM EMPLOYEE) AS a INNER JOIN    (SELECT ID, NAME, SALARY FROM EMPLOYEE) AS b ON a.CHIEF_ID = b.ID    AND b.SALARY = 2*AVG(a.SALARY) ORDER BY b.NAME ASC
